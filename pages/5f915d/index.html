# 手写LazyMan

  • 支持sleep和eat两个方法
  • 支持链式调用
// LazyMan示例
const me = new LazyMan('张三')
me.eat('苹果').eat('香蕉').sleep(5).eat('葡萄')
// 打印
// 张三 eat 苹果
// 张三 eat 香蕉
// 等待5秒
// 张三 eat 葡萄

实现思路

  • 由于有sleep功能，函数不能直接在调用时触发
  • 初始化一个队列，把函数注册进去
  • 由每个item触发next执行（遇到sleep则异步触发，使用setTimeout）
class LazyMan {
  constructor(name) {
    this.name = name
      this.tasks = []
      setTimeout(() => {
        this.next()
      });
  }
  next() {
    const task = this.tasks.shift()
    task && task()
  }
  eat(food) {
    const task = () => {
      console.info(`${this.name} eat ${food}`)
      this.next() // 立刻执行下一个任务
    }
    this.tasks.push(task)
    return this // 链式调用
  }
  sleep(seconds) {
    const task = () => {
      console.info(`${this.name} 开始睡觉`)
      setTimeout(() => {
        console.info(`${this.name} 已经睡完了 ${seconds}s，开始执行下一个任务`)
        this.next() // xx 秒之后再执行下一个任务
      }, seconds * 1000)
  } 
    this.tasks.push(task)
    return this // 链式调用
  }
}

# 手写getType函数

  • 获取详细的数据类型

方法1：

function getType(x) {
  const originType = Object.prototype.toString.call(x) // '[object String]'
  const spaceIndex = originType.indexOf(' ')
  const type = originType.slice(spaceIndex + 1, -1) // 'String' -1不要右边的]
  return type.toLowerCase() // 'string'
}

方法2：

function toType(obj) {
  if (obj == null) return obj + "";
  var reg = /^\[object ([a-zA-Z0-9]+)\]$/i;
  return typeof obj === "object" || typeof obj === "function" ?
    reg.exec(toString.call(obj))[1].toLowerCase() :
    typeof obj;
};

# at

let arr = [1, 2, 4, 5];

/**
 * at - 接收一个整数值并返回该索引对应的元素，允许正数和负数。负整数从数组中的最后一个元素开始倒数
 * @param index
 * @return 匹配给定索引的数组中的元素。如果找不到指定的索引，则返回 undefined
 */
Array.prototype._at = function (index) {
  let n = Math.trunc(index) || 0
  if(n<0) n+=this.length;
  if (n < 0 || n >= this.length) return undefined;
  return this[n];
}
console.log(arr._at(-3.62222222))

# contact

let arr = [1, 2, 4, 5];

/**
 * concat - 不会改变原数组
 * @param args - 数组和/或值，将被合并到一个新的数组中。如果省略了所有 valueN 参数，则 concat 会返回调    
 * 用此方法的现存数组的一个浅拷贝。
 * @return - 返回1个新数组
 */
Array.prototype._contact = function (...args) {

  if(args.length==0) return this;
  let newArray = this;
  for (let i = 0; i < args.length; i++) {
    if(Array.isArray(args[i])){
      newArray.push(...args[i])
    }else{
      newArray.push(args[i])
    }
  }
  return newArray

}

# forEach

let arr = [1, 2, 4, 5];

/**
 * forEach - 不会改变原数组
 * @param {Function} callback - 有3个参数，数组中正在处理的当前元素、数组中正在处理的当前元素的索引、
 * forEach()方法正在操作的数组
 * @param {Any} thisArg - 可选参数。当执行回调函数callback时，用作 this 的值。如果 thisArg 参数有值，则每次 callback函数被    
 * 调用时，this 都会指向 thisArg 参数。如果省略了 thisArg 参数，或者其值为 null 或 undefined，this 则指向全局对象。
 * @return undefined
 */
Array.prototype._forEach = function (callback, thisArg) {
  console.log(thisArg,this);
  // 判断callback是否为函数
  if (typeof callback !== "function") {
    throw new Error(`${callback} is not a function`);
  }
  // 判断当前调用对象是不是数组
  //if (!Array.isArray(this)) {
    //throw new Error(`${this} is not a Array`);
  //}
  for (let i = 0; i < this.length; i++) {
    callback.call(thisArg, this[i], i, this);
  }
};

# filter

let arr = [1, 2, 4, 5];

/**
 * filter - 不会改变原数组
 * @param {Function} callback - 有3个参数，数组中当前正在处理的元素、正在处理的元素在数组中的索引、调     
 * 用了 filter() 的数组本身
 * @param {Any} thisArg - 可选参数。执行 callback 时，用于 this 的值。如果为 filter() 提供一个 
 * thisArg 参数，则它会被作为callback被调用时的 this 值。否则，callback的 this 值在非严格模式下将是全 
 * 局对象，严格模式下为 undefined。
 * @return {Array} - 一个新的、由通过测试的元素组成的数组，如果没有任何数组元素通过测试，则返回空数组。
 */
Array.prototype._filter = function (callback, thisArg) {
  // 判断callback是否为函数
  if (typeof callback !== "function") {
    throw new Error(`callback is not a function`);
  }
  // 存储通过测试的元素
  let newArray = [];
  for (let i = 0; i < this.length; i++) {
    if (callback.call(thisArg, this[i], i, this)) {
      newArray.push(this[i])
    }
  }
  return newArray;
}
let res = arr._filter(function (m) {
  return m>=2
})

# map

let arr = [1, 2, 4, 5];
/**
 * map - 不会改变原数组
 * @param {Function} callback - 有3个参数，数组中当前正在处理的元素、正在处理的元素在数组中的索引、
 * 调用了 filter() 的数组本身
 * @param {Any} thisArg
 * @return {Array} - 一个新数组，每个元素都是回调函数的返回值。
 */
Array.prototype._map = function (callback,thisArg) {
  // 判断callback是否为函数
  if (typeof callback !== "function") {
    throw new Error(`callback is not a function`);
  }
  let newArray = [];
  for (let i = 0; i < this.length; i++) {
    newArray[i] = callback.call(thisArg, this[i], i, this)
  }
  return newArray;
}
let res = arr._map((m)=> ++m)

# every

let arr = [1, 2, 4, 5];

/**
 * every - 不会改变原数组
 * @param callback
 * @param thisArg
 * @return 如果回调函数的每一次返回都为 truthy 值，返回 true，否则返回 false。
 * require('../internals/array-iteration').every
 */
Array.prototype._every = function (callback, thisArg) {
  // 判断callback是否为函数
  if (typeof callback !== "function") {
    throw new Error(`callback is not a function`);
  }
  for (let i = 0; i < this.length; i++) {
    let res = callback.call(thisArg, this[i], i, this);
    if (!res) {
      return false
    }
  }
  return true
}
console.log(arr._every(x => x >= 2))

注意：传入一个空数组，无论如何都会返回 true

# some

/**
 * some - 不会改变原数组
 * @param callback
 * @param thisArg
 * @return 数组中有至少一个元素通过回调函数的测试就会返回 true；所有元素都没有通过回调函数的测试返回值才会为 false。
 */
Array.prototype._some = function (callback, thisArg) {
  // 判断callback是否为函数
  if (Object.prototype.toString.call(callback)  !== "[object Function]") {
    throw new Error(`callback is not a function`);
  }
  for (let i = 0; i < this.length; i++) {
    let res = callback.call(thisArg, this[i], i, this);
    if (res) {
      return true
    }
  }
  return false
}
let res = arr._some(m=>m<2)
console.log(res)

# flat

/**
 * flat - 不会改变原数组
 * @param depth 指定要提取嵌套数组的结构深度，默认值为 1。
 * @return 一个新的数组，其中包含拼接后的子数组元素
 * 
*/
1.不控制嵌套
思路：循环判断数组的每一项是否为数组，是数组就递归调用方法，否则push到新数组里
let arr = [1, [2, [3, [4, 5]]], 6];
function _flat(arr) {
  let result = []
  for (let i = 0; i < arr.length; i++) {
    if(Array.isArray(arr[i])){
      result = result.concat(_flat(arr[i]));
    }else{
      result.push(arr[i])
    }
  }
  return result;
}
//输出：[1, 2, 3, 4, 5, 6]

2.控制嵌套
function _flat(arr, deepth = 1) {
  // 如果是 无穷大，默认20层
  // if (deepth === Infinity) {
  //   deepth = 20;
  // }
  // // 校验参数
  // if (deepth <= 0) {
    //   return nums;
  // }
  let result = [];
  arr.forEach(el => {
  // 控制递归条件
    if (Array.isArray(el) && deepth > 0) {
      result = result.concat(_flat(el, deepth - 1));
    } else {
        result.push(el)
    }
  });
    return result;
}
